def _save(method = :save)
  return self unless @resources

  @client.connection.post(path) do |req|
    req.body = { :tags => @resources.reject(&:destroyed?).map(&:id) }
  end

  true
rescue Faraday::Error::ClientError => e
  if method == :save
    false
  else
    raise e
  end
end
